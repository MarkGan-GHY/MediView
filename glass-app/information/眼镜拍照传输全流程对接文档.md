### 3.4 配置 ImageReader 与预览 Session

```kotlin
@RequiresApi(Build.VERSION_CODES.P)
private fun setUpImageReader(camera: CameraDevice) {
    imageReader?.close()
    // ★ YUV_420_888 格式，buffer 大小 10，1920×1080
    imageReader = ImageReader.newInstance(1920, 1080, ImageFormat.YUV_420_888, 10)

    openTime = -1L

    imageReader!!.setOnImageAvailableListener({ reader ->
        // 过滤开机后前 1 秒的不稳定帧
        if (openTime == -1L) { openTime = System.currentTimeMillis(); return@setOnImageAvailableListener }
        if ((System.currentTimeMillis() - openTime) < 1000L) return@setOnImageAvailableListener

        // ★ acquireLatestImage()：取最新帧，丢弃旧帧（避免处理积压帧）
        val image = reader.acquireLatestImage() ?: return@setOnImageAvailableListener

        if (takePhoto.get()) {
            takePhoto.set(false)                    // 立即复位，防止重复触发
            val bitmap = imageToBitmap(image)       // YUV → Bitmap（见 Step 3）
            image.close()                           // ★ 必须 close，否则 buffer 耗尽相机停止

            bitmap?.let {
                lifecycleScope.launch(Dispatchers.Main) {
                    onPhotoCaptured(it)             // 转到主流程（见 Step 4）
                }
            }
        } else {
            image.close()                           // 即使不拍照也要 close！
        }
    }, backHandler)   // ★ 在后台线程处理，不要传 null

    // 配置 CaptureSession：ImageReader + 两个预览 Surface 三路输出
    val captureRequestBuilder = camera.createCaptureRequest(CameraDevice.TEMPLATE_PREVIEW).apply {
        addTarget(imageReader!!.surface)
        surfaceList.forEach { addTarget(it) }
        set(CaptureRequest.CONTROL_AE_TARGET_FPS_RANGE, Range(5, 10))
    }

    val outputs = listOf(
        OutputConfiguration(imageReader!!.surface),
        OutputConfiguration(surfaceList[0]),
        OutputConfiguration(surfaceList[1])
    )
    val sessionConfig = SessionConfiguration(
        SessionConfiguration.SESSION_REGULAR,
        outputs,
        Executors.newSingleThreadExecutor(),
        object : CameraCaptureSession.StateCallback() {
            override fun onConfigured(session: CameraCaptureSession) {
                session.setRepeatingRequest(captureRequestBuilder.build(), null, null)
                cameraCaptureSession = session
            }
            override fun onConfigureFailed(session: CameraCaptureSession) {}
        }
    )
    camera.createCaptureSession(sessionConfig)
}
```

## 5. Step 3：YUV 图像转 JPEG 字节流

### 5.1 YUV_420_888 → Bitmap（摘自 CameraActivity.imageToBitmap）

```kotlin
/**
 * 将 Camera2 ImageReader 输出的 YUV_420_888 格式 Image 转换为 Bitmap。
 * 路径：YUV_420_888 → NV21 字节数组 → JPEG → Bitmap
 * 无第三方依赖，可直接复制使用。
 *
 * ★ 调用方负责在此函数返回后调用 image.close()
 */
private fun imageToBitmap(image: Image): Bitmap? {
    val planes = image.planes

    val yBuffer = planes[0].buffer   // Y 分量
    val vBuffer = planes[2].buffer   // V 分量（注意：NV21 是 YVU，先取 V）
    val uBuffer = planes[1].buffer   // U 分量

    val ySize = yBuffer.remaining()
    val vSize = vBuffer.remaining()
    val uSize = uBuffer.remaining()

    // 组装 NV21：Y 平面 + 交错的 VU 数据
    val nv21 = ByteArray(ySize + vSize + uSize)
    yBuffer.get(nv21, 0,     ySize)
    vBuffer.get(nv21, ySize, vSize)
    uBuffer.get(nv21, ySize + vSize, uSize)

    val yuvImage = YuvImage(nv21, ImageFormat.NV21, image.width, image.height, null)
    val out = ByteArrayOutputStream()
    yuvImage.compressToJpeg(Rect(0, 0, image.width, image.height), 100, out)
    val jpegBytes = out.toByteArray()
    out.close()

    return BitmapFactory.decodeByteArray(jpegBytes, 0, jpegBytes.size)
}
```

### 5.2 Bitmap → JPEG 字节数组（用于 HTTP 传输）

```kotlin
/**
 * 将 Bitmap 压缩为 JPEG 字节数组，直接作为 HTTP 请求体。
 * quality = 85 在图像质量与传输大小之间取得较好平衡。
 * 1920×1080 图片约 200-400 KB，局域网传输耗时约 50-200 ms。
 */
fun Bitmap.toJpegBytes(quality: Int = 85): ByteArray {
    val out = ByteArrayOutputStream()
    compress(Bitmap.CompressFormat.JPEG, quality, out)
    return out.toByteArray()
}
```

> **注意**：`imageToBitmap()` 内部已经过一次 JPEG 压缩（quality=100），
> `toJpegBytes()` 是第二次压缩。如需最高画质，可在 `imageToBitmap()` 直接返回
> `ByteArray` 跳过 Bitmap 中间步骤，但实现复杂度增加，Demo 阶段无需优化。

## 6. Step 4：HTTP POST 传输到手机

### 6.1 接口参数（来自眼镜端对接技术文档）

| 项目 | 值 |
|------|-----|
| URL | `http://192.168.43.1:8080/analyzeDrug` |
| Method | `POST` |
| Content-Type | `image/jpeg` |
| Body | JPEG 图片原始二进制（**不要 Base64，不要 multipart**） |

### 6.2 OkHttp 实现（必须在 IO 线程执行）

```kotlin
// 在 Activity 顶部作为成员变量，复用同一个 Client 实例
private val httpClient = OkHttpClient.Builder()
    .connectTimeout(5, TimeUnit.SECONDS)
    .readTimeout(15, TimeUnit.SECONDS)
    .build()

private val serverUrl = "http://192.168.43.1:8080/analyzeDrug"

/**
 * 将 JPEG 字节数组 POST 到手机服务，返回解析后的结果。
 * 此函数是 suspend fun，内部切换到 IO 线程，调用方无需关心线程。
 * 返回 null 表示网络异常或服务端出错。
 */
private suspend fun sendToPhone(jpegBytes: ByteArray): DrugResult? =
    withContext(Dispatchers.IO) {
        try {
            val body = jpegBytes.toRequestBody("image/jpeg".toMediaType())
            val request = Request.Builder()
                .url(serverUrl)
                .post(body)
                .build()

            val response = httpClient.newCall(request).execute()

            if (!response.isSuccessful) {
                Log.e("DrugCapture", "HTTP error: ${response.code}")
                return@withContext null
            }

            val json = JSONObject(response.body!!.string())

            if (!json.getBoolean("success")) {
                Log.e("DrugCapture", "Server error: ${json.optString("error")}")
                return@withContext null
            }

            DrugResult(
                drugName    = json.getString("drugName"),
                confidence  = json.getDouble("confidence"),
                warningText = json.getString("warningText"),
                needConfirm = json.getBoolean("needConfirm")
            )
        } catch (e: Exception) {
            Log.e("DrugCapture", "Network exception: ${e.message}")
            null
        }
    }

/** 响应数据类，对应手机端返回的 JSON 字段 */
data class DrugResult(
    val drugName: String,      // 识别出的药物名称
    val confidence: Double,    // 置信度 0.0~1.0
    val warningText: String,   // 直接展示给用户的提示文字
    val needConfirm: Boolean   // true 时需要用户二次确认
)
```

### 6.3 错误处理对照表

| 异常场景 | 原因 | 处理方式 |
|----------|------|---------|
| 连接超时（5s） | 手机未开热点 / 服务未启动 / IP 错误 | `FToast.show("无法连接手机，请检查热点")` |
| HTTP 500 | 手机服务内部出错 | `FToast.show("识别服务异常，请重试")` |
| `success = false` | 服务端逻辑拒绝 | 读取 `error` 字段，记录日志 |
| `null` 返回 | 任何网络/解析异常 | `FToast.show("识别失败，请重试")` |
| `CLEARTEXT not permitted` | 未添加 `usesCleartextTraffic=true` | 检查 Manifest（见第 1.2 节） |

## 7. Step 5：解析响应并双目显示

```kotlin
/**
 * 拍照成功后的主流程。
 * 此函数在主线程被调用（由 onPhotoCaptured 保证）。
 */
private var isProcessing = false    // 防止重复触发

private fun onPhotoCaptured(bitmap: Bitmap) {
    if (isProcessing) return
    isProcessing = true

    // 1. 显示"识别中"提示
    mBindingPair.updateView {
        tvStatus.text = "识别中..."
        tvStatus.visibility = View.VISIBLE
        resultCard.visibility = View.GONE
    }
    FToast.show("正在识别，请稍候")

    // 2. 在协程中执行 IO + 更新 UI
    lifecycleScope.launch {
        // 2a. Bitmap → JPEG 字节（IO 线程）
        val jpegBytes = withContext(Dispatchers.IO) { bitmap.toJpegBytes(quality = 85) }

        // 2b. HTTP POST（IO 线程，sendToPhone 内部已切换）
        val result = sendToPhone(jpegBytes)

        // 2c. 回到主线程更新双目 UI（withContext 已在 Main 线程）
        isProcessing = false

        if (result == null) {
            FToast.show("识别失败，请重试")
            mBindingPair.updateView { tvStatus.visibility = View.GONE }
            return@launch
        }

        // 2d. ★ 双目同时显示结果（必须通过 mBindingPair.updateView）
        mBindingPair.updateView {
            tvStatus.visibility = View.GONE
            resultCard.visibility = View.VISIBLE
            tvDrugName.text    = result.drugName
            tvWarning.text     = result.warningText
            tvConfidence.text  = "置信度：${"%.0f".format(result.confidence * 100)}%"
        }

        // 2e. 若需要二次确认，弹出确认框
        if (result.needConfirm) {
            showConfirmDialog(result)
        }
    }
}
```

                    val s = Surface(cameraPreview.surfaceTexture)
                    surfaceList.add(s); mSurface = s
                    if (surfaceList.size == 2) {
                        lifecycleScope.launch { delay(100); setupCamera2() }
                    }
                }
                override fun onSurfaceTextureDestroyed(st: SurfaceTexture): Boolean {
                    mSurface?.release(); return true
                }
                override fun onSurfaceTextureSizeChanged(st: SurfaceTexture, w: Int, h: Int) {}
                override fun onSurfaceTextureUpdated(st: SurfaceTexture) {}
            }
        }
    }

    @SuppressLint("MissingPermission")
    private fun setupCamera2() {
        cameraManager = getSystemService(Context.CAMERA_SERVICE) as CameraManager
        cameraManager.openCamera(cameraManager.cameraIdList.first(), stateCallback, null)
    }

    private val stateCallback = object : CameraDevice.StateCallback() {
        @RequiresApi(Build.VERSION_CODES.P)
        override fun onOpened(camera: CameraDevice) {
            lifecycleScope.launch {
                cameraDevice = camera; delay(100)
                if (cameraDevice == camera) setUpImageReader(camera)
            }
        }
        override fun onDisconnected(camera: CameraDevice) { cameraDevice = null }
        override fun onError(camera: CameraDevice, error: Int) {}
    }

    @RequiresApi(Build.VERSION_CODES.P)
    private fun setUpImageReader(camera: CameraDevice) {
        imageReader?.close()
        imageReader = ImageReader.newInstance(1920, 1080, ImageFormat.YUV_420_888, 10)
        openTime = -1L

        imageReader!!.setOnImageAvailableListener({ reader ->
            if (openTime == -1L) { openTime = System.currentTimeMillis(); return@setOnImageAvailableListener }
            if (System.currentTimeMillis() - openTime < 1000L) return@setOnImageAvailableListener

            val image = reader.acquireLatestImage() ?: return@setOnImageAvailableListener

            if (takePhoto.get()) {
                takePhoto.set(false)
                val bitmap = imageToBitmap(image)
                image.close()
                bitmap?.let {
                    lifecycleScope.launch(Dispatchers.Main) { onPhotoCaptured(it) }
                }
            } else {
                image.close()
            }
        }, backHandler)

        val builder = camera.createCaptureRequest(CameraDevice.TEMPLATE_PREVIEW).apply {
            addTarget(imageReader!!.surface)
            surfaceList.forEach { addTarget(it) }
            set(CaptureRequest.CONTROL_AE_TARGET_FPS_RANGE, Range(5, 10))
        }
        val outputs = listOf(
            OutputConfiguration(imageReader!!.surface),
            OutputConfiguration(surfaceList[0]),
            OutputConfiguration(surfaceList[1])
        )
        camera.createCaptureSession(SessionConfiguration(
            SessionConfiguration.SESSION_REGULAR, outputs,
            Executors.newSingleThreadExecutor(),
            object : CameraCaptureSession.StateCallback() {
                override fun onConfigured(s: CameraCaptureSession) {
                    s.setRepeatingRequest(builder.build(), null, null)
                    cameraCaptureSession = s
                }
                override fun onConfigureFailed(s: CameraCaptureSession) {}
            }
        ))
    }

    private fun closeCamera() {
        try {
            cameraCaptureSession?.close(); cameraCaptureSession = null
            cameraDevice?.close();         cameraDevice = null
            imageReader?.close();          imageReader = null
            surfaceList.clear()
        } catch (e: Exception) { /* ignore */ }
    }

    // ══════════════════════════════════════════
    //  Step 3：YUV → Bitmap
    // ══════════════════════════════════════════

    private fun imageToBitmap(image: Image): Bitmap? {
        val planes  = image.planes
        val yBuffer = planes[0].buffer
        val vBuffer = planes[2].buffer
        val uBuffer = planes[1].buffer
        val nv21 = ByteArray(yBuffer.remaining() + vBuffer.remaining() + uBuffer.remaining())
        yBuffer.get(nv21, 0,                           yBuffer.remaining())
        vBuffer.get(nv21, yBuffer.capacity(),          vBuffer.remaining())
        uBuffer.get(nv21, yBuffer.capacity() + vBuffer.capacity(), uBuffer.remaining())
        val yuv = YuvImage(nv21, ImageFormat.NV21, image.width, image.height, null)
        val out = ByteArrayOutputStream()
        yuv.compressToJpeg(Rect(0, 0, image.width, image.height), 100, out)
        val bytes = out.toByteArray(); out.close()
        return BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
    }

    private fun Bitmap.toJpegBytes(quality: Int = 85): ByteArray {
        val out = ByteArrayOutputStream()
        compress(Bitmap.CompressFormat.JPEG, quality, out)
        return out.toByteArray()
    }

    // ══════════════════════════════════════════
    //  Step 4：HTTP POST
    // ══════════════════════════════════════════

    private suspend fun sendToPhone(jpegBytes: ByteArray): DrugResult? =
        withContext(Dispatchers.IO) {
            try {
                val request = Request.Builder()
                    .url(serverUrl)
                    .post(jpegBytes.toRequestBody("image/jpeg".toMediaType()))
                    .build()
                val response = httpClient.newCall(request).execute()
                if (!response.isSuccessful) return@withContext null
                val json = JSONObject(response.body!!.string())
                if (!json.getBoolean("success")) return@withContext null
                DrugResult(
                    drugName    = json.getString("drugName"),
                    confidence  = json.getDouble("confidence"),
                    warningText = json.getString("warningText"),
                    needConfirm = json.getBoolean("needConfirm")
                )
            } catch (e: Exception) {
                Log.e("DrugCapture", "send failed: ${e.message}")
                null
            }
        }

    data class DrugResult(
        val drugName: String,
        val confidence: Double,
        val warningText: String,
        val needConfirm: Boolean
    )

    // ══════════════════════════════════════════
    //  Step 5：结果显示
    // ══════════════════════════════════════════

    private fun onPhotoCaptured(bitmap: Bitmap) {
        if (isProcessing) return
        isProcessing = true

        mBindingPair.updateView { tvStatus.text = "识别中..." }
        FToast.show("正在识别，请稍候")

        lifecycleScope.launch {
            val jpegBytes = withContext(Dispatchers.IO) { bitmap.toJpegBytes() }
            val result    = sendToPhone(jpegBytes)
            isProcessing  = false

            if (result == null) {
                FToast.show("识别失败，请重试")
                mBindingPair.updateView { tvStatus.text = "" }
                return@launch
            }

            // ★ 双目同时显示结果
            mBindingPair.updateView {
                tvStatus.text     = ""
                tvDrugName.text   = result.drugName
                tvWarning.text    = result.warningText
                tvConfidence.text = "${"%.0f".format(result.confidence * 100)}%"
            }
        }
    }
}

## 9. 关键陷阱与注意事项

| # | 陷阱 | 正确做法 |
|---|------|---------|
| 1 | `acquireNextImage()` 导致帧积压 | 使用 `acquireLatestImage()`，丢弃旧帧 |
| 2 | `image.close()` 遗漏 | 无论是否拍照都必须 close，否则 ImageReader buffer 耗尽相机停止 |
| 3 | 在 ImageReader 回调中直接更新 UI | 必须 `runOnUiThread {}` 或 `lifecycleScope.launch(Dispatchers.Main)` |
| 4 | OkHttp 在主线程调用 | 必须在 `Dispatchers.IO` 中执行，`sendToPhone` 已封装为 suspend |
| 5 | 只有一个 TextureView 就绪时开相机 | 等 `surfaceList.size == 2` 才调用 `setupCamera2()` |
| 6 | 直接操作单侧 binding | 所有 UI 变更必须通过 `mBindingPair.updateView {}` |
| 7 | 忘记 `usesCleartextTraffic=true` | HTTP 明文请求在 Android 9+ 会被系统拦截 |
| 8 | 忘记 `INTERNET` 权限 | 运行时报 `SecurityException: Permission denied` |
| 9 | `DoubleClick` 未绑定 `finish()` | 平台约定，用户无法退出界面 |
| 10 | `MercurySDK.init()` 在 Activity 中调用 | 必须在 `Application.onCreate()` 中调用 |
